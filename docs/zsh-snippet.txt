############################################
# Modern CLI tools integration (eza, bat,
# git-delta, thefuck, etc.)
############################################

# --- eza: modern ls replacement ---
# Use eza instead of ls, with icons and nicer defaults if available
if command -v eza >/dev/null 2>&1; then
  alias ls='eza --group-directories-first --icons'
  alias ll='eza -lah --group-directories-first --icons'
  alias la='eza -a --group-directories-first --icons'
  alias lt='eza -T --group-directories-first --icons'  # tree view
fi

# --- bat: cat replacement with syntax highlighting ---
if command -v bat >/dev/null 2>&1; then
  alias cat='bat --paging=never'
  alias batp='bat --paging=always'   # explicit paging
fi

# --- tree: directory tree ---
if command -v tree >/dev/null 2>&1; then
  alias tre='tree -C'   # colored tree
fi

# --- git-delta: better git diff/pager ---
if command -v delta >/dev/null 2>&1; then
  # Configure git to use delta as pager (run once; safe to keep here)
  git config --global core.pager "delta"
  git config --global interactive.diffFilter "delta --color-only"
  # Optional nicer defaults (line numbers, syntax highlighting, etc.)
  git config --global delta.features "side-by-side line-numbers decorations"
  git config --global delta.navigate "true"
fi

# --- thefuck: correct previous command ---
# Creates the `fuck` alias: type `fuck` and it suggests/corrects last command.
if command -v thefuck >/dev/null 2>&1; then
  eval "$(thefuck --alias fuck)"
fi
